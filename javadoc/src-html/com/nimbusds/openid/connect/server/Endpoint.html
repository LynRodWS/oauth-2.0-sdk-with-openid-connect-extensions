001    package com.nimbusds.openid.connect.server;
002    
003    
004    import com.nimbusds.openid.connect.http.HTTPRequest;
005    import com.nimbusds.openid.connect.http.HTTPResponse;
006    
007    
008    /**
009     * Authorisation server endpoint.
010     *
011     * <p>Parses incoming {@link com.nimbusds.openid.connect.http.HTTPRequest}s into
012     * OpenID Connect {@link com.nimbusds.openid.connect.messages.Request} 
013     * objects which are then passed to a dedicated {@link RequestHandler} to 
014     * perform the actual back-end processing (e.g. by making a database query).
015     *
016     * <p>If the {@link com.nimbusds.openid.connect.http.HTTPRequest} doesn't parse 
017     * to a valid OpenID Connect 
018     * {@link com.nimbusds.openid.connect.messages.Request} or the request is found 
019     * to be otherwise invalid, the endpoint may choose to return a 
020     * {@link com.nimbusds.openid.connect.http.HTTPResponse} indicating an 
021     * {@link com.nimbusds.openid.connect.messages.ErrorCode error} without the 
022     * {@link RequestHandler} being called.
023     *
024     * <p>To do: Add event handler setter for logging and monitoring purposes.
025     *
026     * @author Vladimir Dzhuvinov
027     * @version 0.9 (2012-05-16)
028     */
029    public interface Endpoint {
030    
031    
032            /**
033             * Sets a handler for OpenID Connect 
034             * {@link com.nimbusds.openid.connect.messages.Request}s received at the 
035             * endpoint.
036             *
037             * @param handler The handler.
038             */
039            public void setRequestHandler(final RequestHandler handler);
040            
041            
042            /**
043             * Gets the handler for OpenID Connect 
044             * {@link com.nimbusds.openid.connect.messages.Request}s received at the
045             * endpoint.
046             *
047             * @return The handler.
048             */
049            public RequestHandler getRequestHandler();
050            
051            
052            /**
053             * Processes an HTTP request at the endpoint.
054             *
055             * @param httpRequest The HTTP request to process.
056             *
057             * @return The resulting HTTP response.
058             */
059            public HTTPResponse process(final HTTPRequest httpRequest);
060    }
