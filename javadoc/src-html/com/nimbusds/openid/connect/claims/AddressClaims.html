020    /**
021     * UserInfo address claims, serialisable to a JSON object.
022     *
023     * <p>Example UserInfo address claims set:
024     *
025     * <pre>
026     * {
027     *   "user_id"     : "248289761001",
028     *   "name"        : "Jane Doe",
029     *   "given_name"  : "Jane",
030     *   "family_name" : "Doe",
031     *   "email"       : "janedoe@example.com",
032     *   "picture"     : "http://example.com/janedoe/me.jpg"
033     * }
034     * </pre>
035     *
036     * <p>See http://openid.net/specs/openid-connect-messages-1_0.html#address_claim
037     *
038     * @author Vladimir Dzhuvinov
039     * @version 0.9 (2012-05-17)
040     */

401                    
402                    if (formatted != null)
403                            o.put("formatted", formatted.getClaimValue());
404                            
405                    if (streetAddress != null)
406                            o.put("street_address", streetAddress.getClaimValue());
407                    
408                    if (locality != null)
409                            o.put("locality", locality.getClaimValue());
410                    
411                    if (region != null)
412                            o.put("region", region.getClaimValue());
413                    
414                    if (postalCode != null)
415                            o.put("postal_code", postalCode.getClaimValue());
416                    
417                    if (country != null)
418                            o.put("country", country.getClaimValue());
419                    
420                    return o;
421            }

424            /**
425             * Parses a UserInfo address claims set from the specified JSON object.
426             *
427             * @param jsonObject The JSON object to parse. Must not be {@code null}.
428             *
429             * @return The UserInfo address claims set.
430             *
431             * @throws ParseException If the JSON object couldn't be parsed to a
432             *                        valid UserInfo address claims set.
433             */
434            public static AddressClaims parse(final JSONObject jsonObject)
435                    throws ParseException {
436                    
437                    AddressClaims ac = new AddressClaims();
438                    
439                    // formatted
440                    UserInfo.Address.Formatted formatted = new UserInfo.Address.Formatted();
441                    
442                    if (jsonObject.containsKey(formatted.getClaimName())) {
443                    
444                            ClaimValueParser.parse(jsonObject, formatted);
445                            jsonObject.remove(formatted.getClaimName());
446                            ac.setFormatted(formatted);
447                    }
448                    
449                    // street_address
450                    UserInfo.Address.StreetAddress streetAddress = new UserInfo.Address.StreetAddress();
451                    
452                    if (jsonObject.containsKey(streetAddress.getClaimName())) {
453                    
454                            ClaimValueParser.parse(jsonObject, streetAddress);
455                            jsonObject.remove(streetAddress.getClaimName());
456                            ac.setStreetAddress(streetAddress);
457                    }
458                    
459                    // locality
460                    UserInfo.Address.Locality locality = new UserInfo.Address.Locality();
461                    
462                    if (jsonObject.containsKey(locality.getClaimName())) {
463                    
464                            ClaimValueParser.parse(jsonObject, locality);
465                            jsonObject.remove(locality.getClaimName());
466                            ac.setLocality(locality);
467                    }
468                    
469                    
470                    // region
471                    UserInfo.Address.Region region = new UserInfo.Address.Region();
472                    
473                    if (jsonObject.containsKey(region.getClaimName())) {
474                    
475                            ClaimValueParser.parse(jsonObject, region);
476                            jsonObject.remove(region.getClaimName());
477                            ac.setRegion(region);
478                    }
479                    
480                    
481                    // postal_code
482                    UserInfo.Address.PostalCode postalCode = new UserInfo.Address.PostalCode();
483                    
484                    if (jsonObject.containsKey(postalCode.getClaimName())) {
485                    
486                            ClaimValueParser.parse(jsonObject, postalCode);
487                            jsonObject.remove(postalCode.getClaimName());
488                            ac.setPostalCode(postalCode);
489                    }
490                    
491                    
492                    // country
493                    UserInfo.Address.Country country = new UserInfo.Address.Country();
494                    
495                    if (jsonObject.containsKey(country.getClaimName())) {
496                    
497                            ClaimValueParser.parse(jsonObject, country);
498                            jsonObject.remove(country.getClaimName());
499                            ac.setCountry(country);
500                    }
501                    
502                    
503                    // Add remaing claims as custom
504                    
505                    Iterator <Map.Entry<String,Object>> it = jsonObject.entrySet().iterator();
506                    
507                    while (it.hasNext()) {
508                    
509                            Map.Entry <String,Object> entry = it.next();
510                            
511                            GenericClaim gc = new GenericClaim(entry.getKey());
512                            gc.setClaimValue(entry.getValue());
513                            
514                            ac.addCustomClaim(gc);
515                    }
516                    
517                    return ac;
518            }
