001    package com.nimbusds.openid.connect.claims;
002    
003    
004    /**
005     * Authentication Context Class Reference (ACR). It identifies the 
006     * authentication context, i.e. the information that the relying party may 
007     * require before it makes an entitlements decision with respect to an 
008     * authentication response. Such context may include, but is not limited to, the 
009     * actual authentication method used or level of assurance such as 
010     * ITU-T X.1254 | ISO/IEC 29115 entity authentication assurance level.
011     *
012     * <p>The ACR is represented by a string or an URL string.
013     *
014     * <p>See ISO/IEC DIS 29115
015     *
016     * @author Vladimir Dzhuvinov
017     * @version 0.9 (2012-04-21)
018     */
019    public class AuthenticationContextClassReference extends StringClaim {
020    
021    
022            /**
023             * ACR level 0.
024             */
025            public static final AuthenticationContextClassReference LEVEL_ZERO = new AuthenticationContextClassReference("0");
026    
027    
028            /**
029             * ACR level 1.
030             */
031            public static final AuthenticationContextClassReference LEVEL_ONE = new AuthenticationContextClassReference("1");
032    
033    
034            /**
035             * ACR level 2.
036             */
037            public static final AuthenticationContextClassReference LEVEL_TWO = new AuthenticationContextClassReference("2");
038    
039    
040            /**
041             * ACR level 3.
042             */
043            public static final AuthenticationContextClassReference LEVEL_THREE = new AuthenticationContextClassReference("3");
044    
045    
046            /**
047             * ACR level 4.
048             */
049            public static final AuthenticationContextClassReference LEVEL_FOUR = new AuthenticationContextClassReference("4");
050    
051    
052            /**
053             * Creates a new unspecified Authentication Context Class Reference 
054             * (ACR).
055             */
056            public AuthenticationContextClassReference() {
057            
058                    super();
059            }
060            
061            
062            /**
063             * Creates a new Authentication Context Class Reference (ACR) with the
064             * specified value.
065             *
066             * @param value The ACR value. Must not be {@code null}.
067             */
068            private AuthenticationContextClassReference(final String value) {
069            
070                    super();
071                    
072                    super.setClaimValue(value);
073            }
074            
075            
076            /**
077             * @inheritDoc
078             *
079             * @return "acr".
080             */
081            public String getClaimName() {
082            
083                    return "acr";
084            }
085    }
