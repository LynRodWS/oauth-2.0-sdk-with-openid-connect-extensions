001    package com.nimbusds.openid.connect.claims;
002    
003    
004    import java.net.MalformedURLException;
005    import java.net.URL;
006    
007    import javax.mail.internet.AddressException;
008    import javax.mail.internet.InternetAddress;
009    
010    import net.minidev.json.JSONObject;
011    
012    import com.nimbusds.openid.connect.ParseException;
013    
014    import com.nimbusds.openid.connect.util.JSONObjectUtils;
015    
016    
017    /**
018     * Parses claim values from a JSON object containing a claims set.
019     *
020     * @author Vladimir Dzhuvinov
021     * @version 0.9 (2012-05-17)
022     */
023    public class ClaimValueParser {
024    
025            
026            /**
027             * Parses a claim value from the specified JSON object. Supports all 
028             * {@link Claim.ValueType claim JSON data types} expect
029             * {@link Claim.ValueType#ARRAY array} and {@link Claim.ValueType#OBJECT object}.
030             *
031             *
032             * @param o     The JSON object containing a member representing the
033             *              parsed claim. Must not be {@code null}.
034             * @param claim The claim. Its value will be set to the value of the 
035             *              JSON member with the matching key (name). Must not be 
036             *              {@code null}.
037             *
038             * @throws ParseException If the JSON object doesn't contain a member
039             *                        with a matching key (name), its value is
040             *                        {@code null} or of unexpected type.
041             */
042            @SuppressWarnings("unchecked")
043            public static void parse(final JSONObject o, final Claim claim) 
044                    throws ParseException {
045                    
046                    final String claimName = claim.getClaimName();
047                    
048                    switch (claim.getClaimValueType()) {
049                    
050                            case BOOLEAN:
051                                    claim.setClaimValue(JSONObjectUtils.getBoolean(o, claimName));
052                                    break;
053                    
054                            case INTEGER:
055                                    claim.setClaimValue(JSONObjectUtils.getInt(o, claimName));
056                                    break;
057                                    
058                            case LONG:
059                                    claim.setClaimValue(JSONObjectUtils.getLong(o, claimName));
060                                    break;
061                                    
062                            case FLOAT:
063                                    claim.setClaimValue(JSONObjectUtils.getFloat(o, claimName));
064                                    break;
065                                    
066                            case DOUBLE:
067                                    claim.setClaimValue(JSONObjectUtils.getDouble(o, claimName));
068                                    break;
069                                    
070                            case STRING:
071                                    claim.setClaimValue(JSONObjectUtils.getString(o, claimName));
072                                    break;
073                            
074                            case URL:
075                                    URL url = null;
076                                    
077                                    try {
078                                            url = new URL(JSONObjectUtils.getString(o, claimName));
079                                            
080                                    } catch (MalformedURLException e) {
081                                    
082                                            throw new ParseException("Invalid URL syntax for claim \"" + claimName + "\"", e);
083                                    }
084                                    
085                                    claim.setClaimValue(url);
086                                    break;
087                                    
088                            case EMAIL:
089                                    InternetAddress email = null;
090                                    final boolean isStrict = true;
091                                    
092                                    try {
093                                            email = new InternetAddress(JSONObjectUtils.getString(o, claimName), isStrict);
094                                            
095                                    } catch (AddressException e) {
096                                    
097                                            throw new ParseException("Invalid email syntax for claim \"" + claimName + "\"", e);
098                                    }
099                                    
100                                    claim.setClaimValue(email);
101                                    break;
102                            
103                            case ARRAY:
104                                    claim.setClaimValue(JSONObjectUtils.getJSONArray(o, claimName));
105                                    break;
106                                    
107                            case OBJECT:
108                                    claim.setClaimValue(JSONObjectUtils.getJSONObject(o, claimName));
109                                    break;
110                            
111                            default:
112                                    throw new ParseException("Unsupported type for claim \"" + claimName + "\"");
113                    }
114            }
115    
116    
117            /**
118             * Prevents instantiation.
119             */
120            private ClaimValueParser() {
121            
122                    // Nothing to do
123            }
124    }
