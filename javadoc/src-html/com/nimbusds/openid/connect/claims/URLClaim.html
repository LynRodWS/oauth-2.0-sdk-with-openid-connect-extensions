001    package com.nimbusds.openid.connect.claims;
002    
003    
004    import java.net.URL;
005    
006    
007    /**
008     * The base abstract class for URL-based claims.
009     *
010     * @author Vladimir Dzhuvinov
011     * @version 0.9 (2012-04-20)
012     */
013    public abstract class URLClaim implements Claim<URL> {
014    
015    
016            /**
017             * The claim value.
018             */
019            private URL value;
020            
021            
022            /**
023             * @inheritDoc
024             *
025             * @return The claim value.
026             */
027            public URL getClaimValue() {
028            
029                    return value;
030            }
031            
032            
033            /**
034             * @inheritDoc
035             */
036            public Claim.ValueType getClaimValueType() {
037            
038                    return Claim.ValueType.URL;
039            }
040            
041            
042            /**
043             * @inheritDoc
044             *
045             * @param value The claim value.
046             *
047             * @throws IllegalArgumentException If the value is {@code null}.
048             */
049            public void setClaimValue(final URL value) {
050            
051                    if (value == null)
052                            throw new IllegalArgumentException("The claim value must not be null");
053                    
054                    this.value = value;
055            }
056            
057            
058            /**
059             * Returns the string representation of this claim.
060             *
061             * @return The string representation.
062             */
063            public String toString() {
064            
065                    return this.getClaimName() + ": " + value.toString();
066            }
067            
068            
069            /**
070             * Overrides {@code Object.hashCode()}.
071             *
072             * @return The object hash code.
073             */
074            public int hashCode() {
075            
076                    return value.hashCode();
077            }
078            
079                    
080            /**
081             * Overrides {@code Object.equals()}.
082             *
083             * @param object The object to compare to.
084             *
085             * @return {@code true} if the objects have the same claim name and 
086             *         value, otherwise {@code false}.
087             */
088            public boolean equals(final Object object) {
089            
090                    return object instanceof URLClaim &&
091                           this.getClaimName().equals(((URLClaim)object).getClaimName()) &&
092                           this.getClaimValue().equals(((URLClaim)object).getClaimValue());
093            }
094    }
