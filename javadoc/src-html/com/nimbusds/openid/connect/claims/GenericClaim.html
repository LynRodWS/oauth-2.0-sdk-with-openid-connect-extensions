001    package com.nimbusds.openid.connect.claims;
002    
003    
004    /**
005     * Generic claim. It can be used to represent an arbitrary claim, e.g. a custom
006     * claim outside the typed claims used in OpenID Connect.
007     *
008     * @author Vladimir Dzhuvinov
009     * @version 0.9 (2012-04-21)
010     */
011    public class GenericClaim implements Claim<Object> {
012    
013            
014            /**
015             * The claim name.
016             */
017            private String name;
018            
019             
020            /**
021             * The claim value.
022             */
023            private Object value = null;
024            
025            
026            /**
027             * The claim value type.
028             */
029            private Claim.ValueType type = null;
030            
031            
032            /**
033             * Creates a new generic claim with the specified name.
034             *
035             * @param name  The claim name. Must not be {@code null}.
036             */ 
037            public GenericClaim(final String name) {
038            
039                    if (name == null)
040                            throw new NullPointerException("The claim name must not be null");
041                    
042                    this.name = name;
043            }
044            
045            
046            /**
047             * Gets the canonical claim name.
048             *
049             * @return The canonical claim name.
050             */
051            public String getClaimName() {
052    
053                    return name;
054            }
055            
056            
057            /**
058             * Sets the claim value.
059             *
060             * @param value The claim value. It must map to one of the supported
061             *              {@link Claim.ValueType claim types}. 
062             *
063             * @throws IllegalArgumentException If the value is {@code null} or its
064             *                                  type is not supported.
065             */
066            public void setClaimValue(final Object value) {
067            
068                    type = Claim.ValueType.resolve(value);
069                    
070                    if (type == null)
071                            throw new IllegalArgumentException("Unexpected claim value type");
072            
073                    this.value = value;
074            }
075            
076            
077            /**
078             * Gets the claim value.
079             *
080             * @return The claim value. If defined (not {@code null}) it matches the 
081             *         type indicated by {@link #getClaimValueType}.
082             */
083            public Object getClaimValue() {
084            
085                    return value;
086            }
087            
088            
089            /**
090             * Gets the claim type.
091             *
092             * @return The claim type, {@code null} if the claim value is not
093             *         specified.
094             */
095            public Claim.ValueType getClaimValueType() {
096            
097                    return type;
098            }
099    }
