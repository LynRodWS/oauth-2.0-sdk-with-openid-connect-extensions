020    /**
021     * ID Token claims, serialisable to a JSON object.
022     *
023     * <p>Example ID Token claims set:
024     *
025     * <pre>
026     * {
027     *   "iss"     : "http://server.example.com",
028     *   "user_id" : "248289761001",
029     *   "aud"     : "s6BhdRkqt3",
030     *   "nonce"   : "n-0S6_WzA2Mj",
031     *   "exp"     : 1311281970,
032     *   "iat"     : 1311280970
033     * }
034     * </pre>
035     *
036     * <p>See http://openid.net/specs/openid-connect-messages-1_0.html#id_token
037     *
038     * @author Vladimir Dzhuvinov
039     * @version 0.9 (2012-05-16)
040     */
041    public class IDTokenClaims extends JSONObjectClaims {
042    
043            
044            /**
045             * The names of the reserved ID Token claims.
046             */
047            private static final Set<String> reservedClaimNames = new HashSet<String>();
048            
049            
050            static {
051                    reservedClaimNames.add("iss");
052                    reservedClaimNames.add("user_id");
053                    reservedClaimNames.add("aud");
054                    reservedClaimNames.add("nonce");
055                    reservedClaimNames.add("exp");
056                    reservedClaimNames.add("iat");
057                    reservedClaimNames.add("acr");
058                    reservedClaimNames.add("auth_time");
059                    reservedClaimNames.add("at_hash");
060                    reservedClaimNames.add("c_hash");
061            }
062            
063    
064            /**
065             * Gets the names of the reserved ID Token claims.
066             *
067             * @return The names of the reserved ID Token claims (read-only set).
068             */
069            public static Set<String> getReservedClaimNames() {
070            
071                    return Collections.unmodifiableSet(reservedClaimNames);
072            }

413            /**
414             * @inheritDoc
415             */
416            public void addCustomClaim(final GenericClaim customClaim) {
417            
418                    if (reservedClaimNames.contains(customClaim.getClaimName()))
419                            throw new IllegalArgumentException("Custom claim name conflicts with reserved claim name: " + customClaim.getClaimName());
420            
421                    customClaims.put(customClaim.getClaimName(), customClaim);
422            }

425            /**
426             * @inheritDoc
427             */
428            public JSONObject toJSONObject() {
429            
430                    JSONObject o = new JSONObject();
431                    
432                    o.put("iss", iss.getClaimValue());
433                    o.put("user_id", userID.getClaimValue());
434                    o.put("aud", aud.getClaimValue());
435                    o.put("exp", exp.getClaimValue());
436                    o.put("iat", iat.getClaimValue());
437                    
438                    if (acr != null)
439                            o.put("acr", acr.toString());
440                    
441                    o.put("nonce", nonce.toString());
442                    
443                    if (authTime != null)
444                            o.put("auth_time", authTime.getClaimValue());
445                    
446                    if (atHash != null)
447                            o.put("at_hash", atHash.getClaimValue());
448                    
449                    if (cHash != null)
450                            o.put("c_hash", cHash.getClaimValue());
451                    
452                    return o;
453            }

456            /**
457             * Parses an ID Token claims set from the specified JSON object.
458             *
459             * @param jsonObject The JSON object to parse. Must not be {@code null}.
460             *
461             * @return The ID Token claims set.
462             *
463             * @throws ParseException If the JSON object cannot be parsed to a valid
464             *                        ID Token claims set.
465             */
466            public static IDTokenClaims parse(final JSONObject jsonObject)
467                    throws ParseException {
468                    
469                    // Get required ID token claims
470                    
471                    Issuer iss = new Issuer();
472                    ClaimValueParser.parse(jsonObject, iss);
473                    jsonObject.remove(iss.getClaimName());
474                    
475                    UserID userID = new UserID();
476                    ClaimValueParser.parse(jsonObject, userID);
477                    jsonObject.remove(userID.getClaimName());
478                    
479                    Audience aud = new Audience();
480                    ClaimValueParser.parse(jsonObject, aud);
481                    jsonObject.remove(aud.getClaimName());
482                    
483                    IssueTime iat = new IssueTime();
484                    ClaimValueParser.parse(jsonObject, iat);
485                    jsonObject.remove(iat.getClaimName());
486                    
487                    Nonce nonce = new Nonce(JSONObjectUtils.getString(jsonObject, "nonce"));
488                    
489                    IDTokenClaims idTokenClaims = new IDTokenClaims(iss, userID, aud, iat, nonce);
490                    
491                    
492                    // Get optional ID token claims
493                    
494                    AuthenticationContextClassReference acr = new AuthenticationContextClassReference();
495                    
496                    if (jsonObject.containsKey(acr.getClaimName())) {
497                            
498                            ClaimValueParser.parse(jsonObject, acr);
499                            jsonObject.remove(acr.getClaimName());
500                            idTokenClaims.setAuthenticationContextClassReference(acr);
501                    }
502                    
503                    
504                    AuthenticationTime authTime = new AuthenticationTime();
505                    
506                    if (jsonObject.containsKey(authTime.getClaimName())) {
507                    
508                            ClaimValueParser.parse(jsonObject, authTime);
509                            jsonObject.remove(authTime.getClaimName());
510                            idTokenClaims.setAuthenticationTime(authTime);
511                    }
512                    
513                    
514                    AccessTokenHash atHash = new AccessTokenHash();
515                    
516                    if (jsonObject.containsKey(atHash.getClaimName())) {
517                    
518                            ClaimValueParser.parse(jsonObject, atHash);
519                            jsonObject.remove(atHash.getClaimName());
520                            idTokenClaims.setAccessTokenHash(atHash);
521                    }
522                    
523                    
524                    CodeHash cHash = new CodeHash();
525                    
526                    if (jsonObject.containsKey(cHash.getClaimName())) {
527                    
528                            ClaimValueParser.parse(jsonObject, cHash);
529                            jsonObject.remove(cHash.getClaimName());
530                            idTokenClaims.setCodeHash(cHash);
531                    }
532                    
533                    
534                    // Add remaing claims as custom
535                    
536                    Iterator <Map.Entry<String,Object>> it = jsonObject.entrySet().iterator();
537                    
538                    while (it.hasNext()) {
539                    
540                            Map.Entry <String,Object> entry = it.next();
541                            
542                            GenericClaim gc = new GenericClaim(entry.getKey());
543                            gc.setClaimValue(entry.getValue());
544                            
545                            idTokenClaims.addCustomClaim(gc);
546                    }
547                    
548                    return idTokenClaims;
549            }
