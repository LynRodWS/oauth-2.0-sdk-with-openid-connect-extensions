001    package com.nimbusds.openid.connect.claims;
002    
003    
004    import java.util.HashMap;
005    import java.util.Map;
006    
007    import net.minidev.json.JSONObject;
008    
009    
010    /**
011     * Claims set serialisable to a JSON object.
012     *
013     * @author Vladimir Dzhuvinov
014     * @version 0.9 (2012-05-16)
015     */
016    public abstract class JSONObjectClaims {
017    
018    
019            /**
020             * Custom (non-reserved) claims, empty if none.
021             */
022            protected Map<String,GenericClaim> customClaims = new HashMap<String,GenericClaim>();
023            
024            
025            /**
026             * Returns a JSON object representation of the claims set.
027             *
028             * @return The JSON object representation.
029             */
030            public abstract JSONObject toJSONObject();
031            
032            
033            /**
034             * Gets the custom (non-reserved) claims of the claims set.
035             *
036             * @return The custom claims, empty map if none.
037             */
038            public Map<String, GenericClaim> getCustomClaims() {
039            
040                    return customClaims;
041            }
042            
043            
044            /**
045             * Adds a custom (non-reserved) claim to the claims set.
046             *
047             * @param customClaim The custom claim to add. Must not be {@code null}.
048             *
049             * @throws IllegalArgumentException If the custom claim name conflicts 
050             *                                  with a reserved claim name.
051             */
052            public abstract void addCustomClaim(final GenericClaim customClaim);
053    }
