001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.net.MalformedURLException;
005    import java.net.URL;
006    import java.util.Collections;
007    import java.util.HashSet;
008    import java.util.Set;
009    
010    import javax.mail.internet.ContentType;
011    
012    import net.minidev.json.JSONObject;
013    
014    import com.nimbusds.openid.connect.ParseException;
015    import com.nimbusds.openid.connect.SerializeException;
016    
017    import com.nimbusds.openid.connect.http.CommonContentTypes;
018    import com.nimbusds.openid.connect.http.HTTPResponse;
019    
020    import com.nimbusds.openid.connect.util.JSONObjectUtils;
021    
022    
023    /**
024     * OAuth 2.0 Access Token error response.
025     *
026     * <p>Legal error codes:
027     *
028     * <ul>
029     *     <li>OAuth 2.0 errors:
030     *         <ul>
031     *             <li>{@link ErrorCode#INVALID_REQUEST}
032     *             <li>{@link ErrorCode#INVALID_CLIENT}
033     *             <li>{@link ErrorCode#INVALID_GRANT}
034     *             <li>{@link ErrorCode#UNAUTHORIZED_CLIENT}
035     *             <li>{@link ErrorCode#UNSUPPORTED_GRANT_TYPE}
036     *             <li>{@link ErrorCode#INVALID_SCOPE}
037     *         </ul>
038     * </ul>
039     *
040     * <p>Example HTTP response:
041     *
042     * <pre>
043     * HTTP/1.1 400 Bad Request
044     * Content-Type: application/json
045     * Cache-Control: no-store
046     * Pragma: no-cache
047     * 
048     * {
049     *  "error": "invalid_request"
050     * }
051     * </pre>
052     *
053     * <p>See http://tools.ietf.org/html/draft-ietf-oauth-v2-26#section-5.2
054     *
055     * @author Vladimir Dzhuvinov
056     * @version 0.9 (2012-05-03)
057     */
058    public class TokenErrorResponse implements ErrorResponse {
059    
060    
061            /**
062             * The legal error codes for an OAuth 2.0 Access Token error response.
063             */
064            private static Set<ErrorCode> legalErrorCodes = new HashSet<ErrorCode>();
065            
066            
067            static {
068                    // OAuth 2.0 errors
069                    legalErrorCodes.add(ErrorCode.INVALID_REQUEST);
070                    legalErrorCodes.add(ErrorCode.INVALID_CLIENT);
071                    legalErrorCodes.add(ErrorCode.INVALID_GRANT);
072                    legalErrorCodes.add(ErrorCode.UNAUTHORIZED_CLIENT);
073                    legalErrorCodes.add(ErrorCode.UNSUPPORTED_GRANT_TYPE);
074                    legalErrorCodes.add(ErrorCode.INVALID_SCOPE);
075            }
076            
077            
078            /**
079             * Gets the legal error codes for an OAuth 2.0 Access Token error 
080             * response.
081             *
082             * @return The legal error codes, as a read-only set.
083             */
084            public static Set<ErrorCode> getLegalErrorCodes() {
085            
086                    return Collections.unmodifiableSet(legalErrorCodes);
087            }
088            
089            
090            /**
091             * The error code.
092             */
093            private ErrorCode errorCode;
094            
095            
096            /**
097             * The URL of a web page that includes additional information about the
098             * error.
099             */
100            private URL errorURI = null;
101            
102            
103            /**
104             * Creates a new OAuth 2.0 Access Token error response.
105             *
106             * @param errorCode   The error code. Must match one of the 
107             *                    {@link #getLegalErrorCodes legal error codes} for
108             *                    an authorisation error response and must not be 
109             *                    {@code null}.
110             * @param errorURI    Optional URI of a web page that includes 
111             *                    information about the error, {@code null} if not
112             *                    specified.
113             *
114             * @throws IllegalArgumentException If the specified error code is not
115             *                                  legal for an authorisation error 
116             *                                  response.
117             */
118            public TokenErrorResponse(final ErrorCode errorCode, final URL errorURI) {
119            
120                    if (errorCode == null)
121                            throw new NullPointerException("The error code must not be null");
122                    
123                    if (! legalErrorCodes.contains(errorCode))
124                            throw new IllegalArgumentException("Illegal error code");
125                            
126                    this.errorCode = errorCode;
127                    
128                    this.errorURI = errorURI;
129            }
130            
131    
132            /**
133             * @inheritDoc
134             */
135            public ErrorCode getErrorCode() {
136            
137                    return errorCode;
138            }
139            
140            
141            /**
142             * @inheritDoc
143             */
144            public URL getErrorURI() {
145            
146                    return errorURI;
147            }
148            
149            
150            /**
151             * Returns the JSON object for this token error response.
152             *
153             * @return The JSON object for this token error response.
154             */
155            public JSONObject toJSONObject() {
156            
157                    JSONObject o = new JSONObject();
158                    o.put("error", errorCode.getCode());
159                    o.put("error_description", errorCode.getDescription());
160                    
161                    if (errorURI != null)
162                            o.put("error_uri", errorURI.toString());
163                    
164                    return o;
165            }
166            
167            
168            /**
169             * @inheritDoc
170             */
171            public HTTPResponse toHTTPResponse()
172                    throws SerializeException {
173                    
174                    // HTTP status 400
175                    HTTPResponse httpResponse = new HTTPResponse(HTTPResponse.SC_BAD_REQUEST);
176                    
177                    httpResponse.setContentType(CommonContentTypes.APPLICATION_JSON);
178                    httpResponse.setCacheControl("no-store");
179                    httpResponse.setPragma("no-cache");
180                    
181                    httpResponse.setContent(toJSONObject().toString());
182                    
183                    return httpResponse;
184            }
185            
186            
187            /**
188             * Parses an OAuth 2.0 Token Error response.
189             *
190             * @param httpResponse The HTTP response to parse. Must not be 
191             *                     {@code null}.
192             *
193             * @throws ParseException If the HTTP response cannot be parsed to a 
194             *                        valid OAuth 2.0 Token Error response.
195             */
196            public static TokenErrorResponse parse(final HTTPResponse httpResponse)
197                    throws ParseException {
198                    
199                    if (httpResponse.getStatusCode() != HTTPResponse.SC_BAD_REQUEST)
200                           throw new ParseException("Unexpected HTTP status code, must be 400 (Bad request): " + httpResponse.getStatusCode());
201    
202                    // Cache-Control and Pragma headers are ignored
203                    
204                    ContentType contentType = httpResponse.getContentType();
205                    
206                    if (contentType == null)
207                            throw new ParseException("Missing HTTP Content-Type header");
208                    
209                    if (! contentType.match(CommonContentTypes.APPLICATION_JSON))
210                            throw new ParseException("Expected HTTP Content-Type \"" + CommonContentTypes.APPLICATION_JSON + "\"");
211                    
212                    String content = httpResponse.getContent();
213                    
214                    if (content == null)
215                            throw new ParseException("Missing HTTP content");
216                    
217                    JSONObject jsonObject = null;
218                    
219                    try {
220                            jsonObject = JSONObjectUtils.parseJSONObject(content);
221                    
222                    } catch (ParseException e) {
223                    
224                            throw new ParseException("Invalid JSON object: " + e.getMessage(), e);
225                    }
226                    
227                    String errorCodeString = null;
228                    ErrorCode errorCode = null;
229                    URL errorURI = null;
230                    
231                    try {
232                            errorCodeString = JSONObjectUtils.getString(jsonObject, "error");
233                            errorCode = ErrorCode.valueOf(errorCodeString.toUpperCase());
234                            
235                            if (jsonObject.containsKey("error_uri"))
236                                    errorURI = new URL(JSONObjectUtils.getString(jsonObject, "error_uri"));
237                            
238                    } catch (ParseException e) {
239                    
240                            throw new ParseException("Missing or invalid token error response parameter: " + e.getMessage(), e);
241                            
242                    } catch (IllegalArgumentException e) {
243                    
244                            throw new ParseException("Invalid error code: " + errorCodeString, e);
245                    
246                    } catch (MalformedURLException e) {
247                    
248                            throw new ParseException("Invalid error URI: " + e.getMessage(), e);
249                    }
250                    
251                    
252                    if (! getLegalErrorCodes().contains(errorCode))
253                            throw new ParseException("Illegal token response error code: " + errorCode.getCode());
254                    
255                    return new TokenErrorResponse(errorCode, errorURI);
256            }
257    }
