759            /**
760             * Downloads a JWT-encoded OpenID request object at the specified URL.
761             *
762             * @param url The request object URL. Must not be {@code null}.
763             *
764             * @return The downloaded JWT-encoded OpenID request object.
765             *
766             * @throws IOException  If the HTTP connection to the specified URL 
767             *                      failed.
768             * @throws JWTException If the content at the specified URL couldn't be
769             *                      parsed to a valid JSON Web Token (JWT).
770             */
771            protected static JWT downloadRequestObject(final URL url)
772                    throws IOException, JWTException {
773                    
774                    HttpURLConnection con = (HttpURLConnection)url.openConnection();
775                    
776                    StringBuilder sb = new StringBuilder();
777                    
778                    BufferedReader input = new BufferedReader(new InputStreamReader(con.getInputStream()));
779                    
780                    String line = null;
781                    
782                    while ((line = input.readLine()) != null) {
783                    
784                            sb.append(line);
785                            sb.append(System.getProperty("line.separator"));
786                    }
787                    
788                    input.close();
789                    
790                    // Save HTTP code + message
791                    final int statusCode = con.getResponseCode();
792                    final String statusMessage = con.getResponseMessage();
793                    
794                    return JWT.parse(sb.toString());
795            }

798            /**
799             * Resolves the OpenID request object (if any).
800             *
801             * @throws ResolveException For a request object URI that couldn't be
802             *                          resolved to a valid JWT.
803             */
804            private void resolveRequestObject()
805                    throws ResolveException {
806            
807                    // Do we have a request object?
808                    if (! this.hasRequestObject())
809                            return;
810            
811                    if (requestObject != null) {
812                    
813                            resolvedRequestObject = requestObject;
814                    }
815                    else if (requestURI != null) {
816                    
817                            try {
818                                    resolvedRequestObject = downloadRequestObject(requestURI);
819                                    
820                            } catch (IOException e) {
821                            
822                                    throw new ResolveException("Couldn't resolve request object: " + e.getMessage(), e);
823                                    
824                            } catch (JWTException e) {
825                            
826                                    throw new ResolveException("Couldn't resolve request object: " + e.getMessage(), e);
827                            }
828                    }
829            }

835             * <p>The OpenID request object is resolved as follows:
836             *
837             * <ul>
838             *     <li>If no request object is specified this method returns 
839             *         {@code null}.
840             *     <li>If the request object is specified directly by a 
841             *         {@code request} parameter this method returns the same value 
842             *         as {@link #getRequestObject}.
843             *     <li>If the request object is specified by a {@code request_uri} 
844             *         parameter this method returns the referenced JWT.
845             * </ul>
846             *
847             * @return The resolved request object, {@code null} if not specified.
848             *
849             * @throws ResolveException For a request object URI that couldn't be
850             *                          resolved to a valid JWT.
851             */
852            public JWT getResolvedRequestObject()
853                    throws ResolveException {
854            
855                    if (! this.hasRequestObject())
856                            return null;
857                    
858                    // Cached object?
859                    if (resolvedRequestObject == null)
860                            resolveRequestObject();
861                    
862                    return resolvedRequestObject;
863            }

866            /**
867             * Gets the JSON object of the resolved OpenID request object.
868             *
869             * @return The resolved OpenID request object JSON.
870             *
871             * @throws ResolveException If resolution failed, the JWT is not in the
872             *                          required state (verified for signed JWTs or
873             *                          decrypted for encrypted JWTs) or the request
874             *                          object JSON is invalid.
875             */
876            private JSONObject getResolvedRequestObjectJSON()
877                    throws ResolveException {
878                    
879                    if (requestObjectJSON == null) {
880                    
881                            JWT jwt = getResolvedRequestObject();
882                            
883                            if (jwt instanceof SignedJWT && 
884                                ((SignedJWT)jwt).getState() != SignedJWT.State.VERIFIED)
885                                    throw new ResolveException("The request object JWT is signed and must be verified first");
886                            
887                            else if (jwt instanceof EncryptedJWT &&
888                                     ((EncryptedJWT)jwt).getState() != EncryptedJWT.State.DECRYPTED)
889                                    throw new ResolveException("The request object JWT is encrypted and must be decrypted first");
890                            
891                            requestObjectJSON = jwt.getClaimsSet().toJSONObject();
892                            
893                            if (requestObjectJSON == null)
894                                    throw new ResolveException("Invalid request object (JWT) JSON");
895                    }
896                    
897                    // Validate existence of mandatory response_type and scope
898                    if (! requestObjectJSON.containsKey("response_type"))
899                            throw new ResolveException("Missing \"response_type\" parameter in the request object (JWT)");
900                    
901                    if (! requestObjectJSON.containsKey("scope"))
902                            throw new ResolveException("Missing \"scope\" parameter in the request object (JWT)");
903                    
904                    return requestObjectJSON;
905            }

978            /**
979             * Parses an authorisation request from the specified URL query string.
980             *
981             * <p>Example URL query string:
982             *
983             * <pre>
984             * ?response_type=token%20id_token
985             * &client_id=s6BhdRkqt3
986             * &redirect_uri=https%3A%2F%2Fclient.example.com%2Fcb
987             * &scope=openid%20profile
988             * &state=af0ifjsldkj
989             * &nonce=n-0S6_WzA2Mj
990             * </pre>
991             *
992             * @param query The URL query string. Must not be {@code null}.
993             *
994             * @return The parsed authorisation request.
995             *
996             * @throws ParseException If the query string couldn't be parsed to a 
997             *                        valid authorisation request.
998             */
999            public static AuthorizationRequest parse(final String query)
1000                    throws ParseException {
1001                    
1002                    if (query == null)
1003                            throw new ParseException("Missing URL query string");
1004            
1005                    Map <String,String> params = URLUtils.parseParameters(query);
1006                    
1007                    String v = null;
1008                    
1009                    // Mandatory params
1010                    
1011                    v = params.get("response_type");
1012                    
1013                    if (v == null || v.trim().isEmpty())
1014                            throw new ParseException("Missing \"response_type\" parameter");
1015                    
1016                    ResponseTypeSet rts = null;
1017                    
1018                    try {
1019                            rts = ResponseTypeSet.parse(v);
1020                    
1021                    } catch (ParseException e) {
1022                            
1023                            throw new ParseException("Invalid \"response_type\" parameter: " + e.getMessage(), e);
1024                    }
1025                            
1026                    v = params.get("scope");
1027                    
1028                    if (v == null || v.trim().isEmpty())
1029                            throw new ParseException("Missing \"scope\" parameter");
1030                    
1031                    Scope scope = null;
1032                    
1033                    try {
1034                            scope = Scope.parseStrict(v);
1035                            
1036                    } catch (ParseException e) {
1037                    
1038                            throw new ParseException("Invalid \"scope\" parameter: " + e.getMessage(), e);
1039                    }
1040                    
1041                    
1042                    v = params.get("client_id");
1043                    
1044                    if (v == null || v.trim().isEmpty())
1045                            throw new ParseException("Missing \"client_id\" parameter");
1046                    
1047                    ClientID clientID = new ClientID();
1048                    clientID.setClaimValue(v);
1049                    
1050                    
1051                    v = params.get("redirect_uri");
1052                    
1053                    if (v == null || v.trim().isEmpty())
1054                            throw new ParseException("Missing \"redirect_uri\" parameter");
1055                            
1056                    URL redirectURI = null;
1057                    
1058                    try {
1059                            redirectURI = new URL(v);
1060                            
1061                    } catch (MalformedURLException e) {
1062                    
1063                            throw new ParseException("Invalid \"redirect_uri\" parameter: " + e.getMessage(), e);
1064                    }
1065                    
1066                    v = params.get("nonce");
1067                    
1068                    if (v == null || v.trim().isEmpty())
1069                            throw new ParseException("Missing \"nonce\" parameter");
1070                    
1071                    Nonce nonce = new Nonce(v);
1072                    
1073                    
1074                    AuthorizationRequest request = new AuthorizationRequest(rts, scope, clientID, redirectURI, nonce);
1075                    
1076                    
1077                    // Optional params
1078                    
1079                    v = params.get("state");
1080                    
1081                    if (v != null && ! v.trim().isEmpty())
1082                            request.setState(new State(v));
1083                    
1084                    
1085                    v = params.get("display");
1086                    
1087                    if (v != null && ! v.trim().isEmpty()) {
1088                    
1089                            try {
1090                                    request.setDisplay(Display.parse(v));
1091                                    
1092                            } catch (ParseException e) {
1093                                    
1094                                    throw new ParseException("Invalid \"display\" parameter: " + e.getMessage(), e);
1095                            }
1096                    }
1097                    
1098                    v = params.get("prompt");
1099                    
1100                    if (v != null && ! v.trim().isEmpty()) {
1101                    
1102                            try {
1103                                    request.setPrompt(Prompt.parse(v));
1104                                    
1105                            } catch (ParseException e) {
1106                            
1107                                    throw new ParseException("Invalid \"prompt\" parameter: " + e.getMessage(), e);
1108                            }
1109                    }
1110                    
1111                    
1112                    v = params.get("request");
1113                    
1114                    if (v != null && ! v.trim().isEmpty()) {
1115                    
1116                            try {
1117                                    request.setRequestObject(JWT.parse(v));
1118                                    
1119                            } catch (JWTException e) {
1120                            
1121                                    throw new ParseException("Invalid \"request\" parameter: " + e.getMessage(), e);
1122                            }
1123                    }
1124                    
1125                    v = params.get("request_uri");
1126                    
1127                    if (v != null && ! v.trim().isEmpty()) {
1128            
1129                            // request_object and request_uri must not be defined at the same time
1130                            if (request.hasRequestObject())
1131                                    throw new ParseException("Invalid request: Found mutually exclusive \"request_object\" and \"request_uri\"");
1132            
1133                            try {
1134                                    request.setRequestObjectURI(new URL(v));
1135                    
1136                            } catch (MalformedURLException e) {
1137                            
1138                                    throw new ParseException("Invalid \"redirect_uri\" parameter", e);
1139                            }
1140                    }
1141            
1142                    return request;
1143            }
