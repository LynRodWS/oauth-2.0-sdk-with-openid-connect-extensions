001    package com.nimbusds.openid.connect.messages;
002    
003    
004    /**
005     * A member in the {@link Scope} parameter of an {@link AuthorizationRequest}.
006     *
007     * @author Vladimir Dzhuvinov
008     * @version 0.9 (2012-04-09)
009     */
010    public interface ScopeMember {
011    
012    
013            /**
014             * Enumeration of the {@link ScopeMember scope member} requirement 
015             * types.
016             */
017            public static enum Type {
018            
019                    /**
020                     * The member must be present in the {@link Scope} parameter.
021                     */
022                    REQUIRED,
023                    
024                    
025                    /**
026                     * The member may be optionally included in the {@link Scope}
027                     * parameter.
028                     */
029                    OPTIONAL;
030            }
031            
032    
033            /**
034             * Returns the string identifier of the scope member.
035             *
036             * @return The string identifier of the scope member.
037             */
038            public String toString();
039            
040            
041            /**
042             * Returns the requirement type of the scope member.
043             *
044             * @return The requirement type of the scope member.
045             */
046            public Type getType();
047    }
