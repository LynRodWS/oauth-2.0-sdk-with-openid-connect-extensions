001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import com.nimbusds.openid.connect.ParseException;
005    
006    
007    /**
008     * Enumeration of the OAuth 2.0 grant types used by OpenID Connect.
009     *
010     * <p>See draft-ietf-oauth-v2-26
011     *
012     * @author Vladimir Dzhuvinov
013     * @version 0.9 (2012-05-11)
014     */
015    public enum GrantType {
016    
017            
018            /**
019             * Authorisation code.
020             */
021            AUTHORIZATION_CODE,
022            
023            
024            /**
025             * Refresh token.
026             */
027            REFRESH_TOKEN;
028            
029            
030            /**
031             * Returns the canonical string representation of this grant type.
032             *
033             * @return The canonical string representation.
034             */
035            public String toString() {
036            
037                    return super.toString().toLowerCase();
038            }
039            
040            
041            /**
042             * Parses a grant type.
043             *
044             * @param s A canonical string representation of a grant type. Must not
045             *          be {@code null}.
046             *
047             * @return The parsed grant type.
048             *
049             * @throws ParseException If the string doesn't match a grant type.
050             */
051            public static GrantType parse(final String s)
052                    throws ParseException {
053            
054                    if (s == null || s.trim().isEmpty())
055                            throw new ParseException("Couldn't parse grant type: Null or empty string");
056                    
057                    if (s.equals("authorization_code"))
058                            return AUTHORIZATION_CODE;
059                            
060                    else if (s.equals("refresh_token"))
061                            return REFRESH_TOKEN;
062                            
063                    else
064                            throw new ParseException("Couldn't parse grant type: Unexpected grant type: " + s);
065            }
066    }
