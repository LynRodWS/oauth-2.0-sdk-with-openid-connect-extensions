001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.util.Map;
005    
006    import com.nimbusds.openid.connect.ParseException;
007    
008    import com.nimbusds.openid.connect.http.CommonContentTypes;
009    import com.nimbusds.openid.connect.http.HTTPRequest;
010    
011    import com.nimbusds.openid.connect.util.URLUtils;
012    
013    
014    /**
015     * The base abstract class for access token and refresh token requests to the
016     * Token endpoint. The request type can be inferred by calling 
017     * {@link #getGrantType}.
018     *
019     * <p>Example access token request:
020     *
021     * <pre>
022     * POST /token HTTP/1.1
023     * Host: server.example.com
024     * Authorization: Basic czZCaGRSa3F0MzpnWDFmQmF0M2JW
025     * Content-Type: application/x-www-form-urlencoded;charset=UTF-8
026     *
027     * grant_type=authorization_code&code=SplxlOBeZQQYbYS6WxSbIA
028     * &redirect_uri=https%3A%2F%2Fclient%2Eexample%2Ecom%2Fcb
029     * </pre>
030     *
031     * <p>See draft-ietf-oauth-v2-26
032     *
033     * @author Vladimir Dzhuvinov
034     * @version 0.9 (2012-05-11)
035     */
036    public abstract class TokenRequest implements Request {
037    
038    
039            /**
040             * The grant type.
041             */
042            private GrantType grantType;
043            
044            
045            /**
046             * The client authentication, {@code null} if none.
047             */
048            private ClientAuthentication clientAuth;
049            
050            
051            /**
052             * Creates a new token request.
053             *
054             * @param grantType  The grant type. Must not be {@code null}.
055             * @param clientAuth The client authentication, {@code null} if none.
056             */
057            protected TokenRequest(final GrantType grantType, final ClientAuthentication clientAuth) {
058            
059                    if (grantType == null)
060                            throw new NullPointerException("The grant type must not be null");
061                    
062                    this.grantType = grantType;
063                    
064                    this.clientAuth = clientAuth;
065            }
066            
067            
068            /**
069             * Gets the grant type.
070             *
071             * @return The grant type.
072             */
073            public GrantType getGrantType() {
074            
075                    return grantType;
076            }
077            
078            
079            /**
080             * Gets the client authentication.
081             *
082             * @return The client authentication, {@code null} if none.
083             */
084            public ClientAuthentication getClientAuthentication() {
085            
086                    return clientAuth;
087            }
088            
089            
090            /**
091             * Parses the specified HTTP request for a token request.
092             *
093             * @param httpRequest The HTTP request. Must not be {@code null}.
094             *
095             * @return The token request.
096             *
097             * @throws ParseException If the HTTP request couldn't be parsed to a 
098             *                        valid token request.
099             */
100            public static TokenRequest parse(final HTTPRequest httpRequest)
101                    throws ParseException {
102                    
103                    // Only HTTP POST accepted
104                    httpRequest.ensureMethod(HTTPRequest.Method.POST);
105                    httpRequest.ensureContentType(CommonContentTypes.APPLICATION_URLENCODED);
106                    
107                    // No fragment!
108                    // May use query component!
109                    
110                    String query = httpRequest.getQuery();
111                    
112                    if (query == null)
113                            throw new ParseException("Missing HTTP POST request entity body");
114                    
115                    Map<String,String> params = URLUtils.parseParameters(query);
116                    
117                    
118                    // Parse grant type
119                    final String grantTypeString = params.get("grant_type");
120                    
121                    if (grantTypeString == null)
122                            throw new ParseException("Missing \"grant_type\" parameter");
123                    
124                    GrantType grantType = GrantType.parse(grantTypeString);
125                    
126                    switch (grantType) {
127                    
128                            case AUTHORIZATION_CODE:
129                                    return AccessTokenRequest.parse(httpRequest);
130                            
131                            case REFRESH_TOKEN:
132                                    return RefreshTokenRequest.parse(httpRequest);
133                            
134                            default:
135                                    throw new ParseException("Unsupported \"grant_type\": " + grantType);
136                    }
137            }
138    }
