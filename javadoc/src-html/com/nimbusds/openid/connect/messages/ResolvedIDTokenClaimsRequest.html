001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.util.LinkedList;
005    import java.util.List;
006    
007    import net.minidev.json.JSONArray;
008    import net.minidev.json.JSONObject;
009    
010    import com.nimbusds.langtag.LangTagException;
011    
012    
013    /**
014     * The resolved ID Token claims request.
015     *
016     * @author Vladimir Dzhuvinov
017     * @version 0.9 (2012-04-25)
018     */
019    public class ResolvedIDTokenClaimsRequest extends ClaimsRequest {
020    
021    
022            /**
023             * The maximum authentication age, in seconds. -1 if not specified.
024             */
025            private int maxAge = -1;
026            
027            
028            /**
029             * Gets the default ID Token claims request.
030             *
031             * @return The default ID Token claims request.
032             */
033            public static JSONObject getDefaultClaimsRequest() {
034            
035                    JSONObject defaultClaims = new JSONObject();
036                    
037                    defaultClaims.put("iss", null);
038                    defaultClaims.put("user_id", null);
039                    defaultClaims.put("aud", null);
040                    defaultClaims.put("exp", null);
041                    defaultClaims.put("iat", null);
042                    defaultClaims.put("nonce", null);
043                    
044                    return defaultClaims;
045            }
046            
047            
048            /**
049             * Creates a new resolved ID Token claims request.
050             *
051             * @param idTokenObject The IDToken object from the optional OpenID
052             *                      request object. {@code null} if not specified.
053             *
054             * @throws ResolveException If the ID Token claims request couldn't be
055             *                          resolved.
056             */
057            public ResolvedIDTokenClaimsRequest(final JSONObject idTokenObject)
058                    throws ResolveException {
059            
060                    claims.putAll(getDefaultClaimsRequest());
061                    
062                    if (idTokenObject != null) {
063                    
064                            if (idTokenObject.containsKey("claims") &&
065                                idTokenObject.get("claims") instanceof JSONObject) {
066                    
067                                    // Merge claims
068                                    JSONObject additionalClaims = (JSONObject)idTokenObject.get("claims");
069    
070                                    claims.putAll(additionalClaims);
071                            }
072                            
073                            try {
074                                    preferredLocales = ClaimsRequest.parsePreferredLocales(idTokenObject);
075                                    
076                            } catch (LangTagException e) {
077                            
078                                    throw new ResolveException("Couldn't parse preferred locales: " + e.getMessage(), e);
079                            }
080                            
081                            
082                            // Parse max_age
083                            if (idTokenObject.get("max_age") != null &&
084                                idTokenObject.get("max_age") instanceof Number)
085                                    maxAge = ((Number)idTokenObject.get("max_age")).intValue();
086                    }
087            }
088            
089            
090            /**
091             * Gets the required user ID (shorthand method).
092             *
093             * <p>Example claim structure:
094             *
095             * <pre>
096             * { "user_id": {"value":"248289761001"}, ... }
097             * </pre>
098             *
099             * @return The required user ID, {@code null} if not specified.
100             *
101             * @throws ResolveException
102             */
103            public String getUserID()
104                    throws ResolveException {
105            
106                    Object uidObject = claims.get("user_id");
107                    
108                    if (uidObject == null)
109                            return null;
110                            
111                    if (! (uidObject instanceof JSONObject))
112                            throw new ResolveException("Unexpected \"user_id\" type, must be a JSON object");
113                            
114                    Object uidValue = ((JSONObject)uidObject).get("value");
115                    
116                    if (uidValue == null)
117                            return null;
118    
119                    if (! (uidValue instanceof String))
120                            throw new ResolveException("Unexpected \"value\" type, must be a JSON string");
121                            
122                    return (String)uidValue;
123            }
124            
125            
126            /**
127             * Gets the required Authentication Context Class References (ACRs) 
128             * (shorthand method).
129             *
130             * <p>Example claim structure:
131             *
132             * <pre>
133             * { "acr": {"values":["2","http://id.incommon.org/assurance/bronze"]}, ... }
134             * </pre>
135             *
136             * @return The required ACRs, {@code null} if not specified.
137             *
138             * @throws ResolveException
139             */
140            public String[] getAuthenticationContextClassReference()
141                    throws ResolveException {
142            
143                    Object acrObject = claims.get("acr");
144                    
145                    if (acrObject == null)
146                            return null;
147                            
148                    if (! (acrObject instanceof JSONObject))
149                            throw new ResolveException("Unexpected \"acr\" type, must be a JSON object");
150            
151                    Object acrValues = ((JSONObject)acrObject).get("values");
152                    
153                    if (acrValues == null)
154                            return null;
155                            
156                    if (! (acrValues instanceof JSONArray))
157                            throw new ResolveException("Unexpected \"acr\" values type, must be a JSON array");
158            
159                    int numElements = ((List)acrValues).size();
160            
161                    String[] acr = new String[numElements];
162                    
163                    for (int i=0; i < numElements; i++) {
164                    
165                            if (! (((List)acrValues).get(i) instanceof String))
166                                    throw new ResolveException("Unexpected ACR value, must be a JSON string");
167                            
168                            acr[i] = (String)((List)acrValues).get(i);
169                    }
170                    
171                    return acr;
172            }
173            
174            
175            /**
176             * Gets the expected max authentication age (shorthand method).
177             *
178             * @return The maximum authentication age, in seconds. -1 if not 
179             *         specified.
180             */
181            public int getMaxAge()
182                    throws ResolveException {
183            
184                    return maxAge;
185            }
186    }
