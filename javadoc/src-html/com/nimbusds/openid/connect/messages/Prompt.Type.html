001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.util.HashSet;
005    import java.util.Iterator;
006    
007    import com.nimbusds.openid.connect.ParseException;
008    
009    
010    /**
011     * Prompts for end-user reauthentication and consent.
012     *
013     * @author Vladimir Dzhuvinov
014     * @version 0.9 (2012-04-22)
015     */
016    public class Prompt extends HashSet<Prompt.Type> {
017    
018    
019            /**
020             * Enumeration of the prompt types.
021             */
022            public static enum Type {
023            
024            
025                    /** 
026                     * The authorization server must not display any authentication 
027                     * or consent UI pages. An error is returned if the end user is 
028                     * not already authenticated or the client does not have 
029                     * pre-configured consent for the requested scopes. This can be 
030                     * used as a method to check for existing authentication and/or 
031                     * consent.
032                     */
033                    NONE,
034    
035    
036                    /**
037                     * The authorisation server must prompt the end user for 
038                     * reauthentication.
039                     */
040                    LOGIN,
041    
042    
043                    /**
044                     * The authorisation server must prompt the end-user for consent 
045                     * before returning information to the client.
046                     */
047                    CONSENT,
048    
049    
050                    /**
051                     * The authorisation server must prompt the end-user to select a 
052                     * user account. This allows a user who has multiple accounts at 
053                     * the authorization server to select amongst the multiple 
054                     * accounts that they may have current sessions for.
055                     */
056                    SELECT_ACCOUNT;
057                    
058                    
059                    /**
060                     * Returns the canonical string representation of this prompt
061                     * type.
062                     *
063                     * @return The canonical string representation.
064                     */
065                    public String toString() {
066                    
067                            return super.toString().toLowerCase();
068                    }
069                    
070                    
071                    /**
072                     * Parses a prompt type.
073                     *
074                     * @param s The string to parse.
075                     *
076                     * @return The parsed prompt type.
077                     *
078                     * @throws ParseException If the parsed string is {@code null} 
079                     *                        or doesn't match a prompt type.
080                     */
081                    public static Type parse(final String s)
082                            throws ParseException {
083    
084                            if (s == null || s.trim().isEmpty())
085                                    throw new ParseException("Couldn't parse prompt type: Null or empty value");
086    
087                            if (s.equals("none"))
088                                    return NONE;
089    
090                            else if (s.equals("login"))
091                                    return LOGIN;
092    
093                            else if (s.equals("consent"))
094                                    return CONSENT;
095    
096                            else if (s.equals("select_account"))
097                                    return SELECT_ACCOUNT;
098    
099                            else
100                                    throw new ParseException("Couldn't parse prompt type: Unexpected prompt type: " + s);
101                    }
102            }
103            
104            
105            /**
106             * Creates a new empty prompt set.
107             */
108            public Prompt() {
109            
110                    // Nothing to do
111            }
112            
113            
114            /**
115             * Checks if the prompt set is valid. This is done by examining the set
116             * for a conflicting {@link Type#NONE} value.
117             *
118             * @return {@code true} if this prompt set if valid, else {@code false}.
119             */
120            public boolean isValid() {
121            
122                    if (size() > 1 && contains(Type.NONE))
123                            return false;
124                    else
125                            return true;
126            }
127            
128            
129            /**
130             * Returns the canonical string representation of this prompt set. The 
131             * values are delimited by space.
132             *
133             * @return The string representation of this scope.
134             */
135            public String toString() {
136            
137                    StringBuilder sb = new StringBuilder();
138            
139                    Iterator<Type> it = super.iterator();
140                    
141                    while (it.hasNext()) {
142                    
143                            sb.append(it.next().toString());
144                            
145                            if (it.hasNext())
146                                    sb.append(" ");
147                    }
148            
149                    return sb.toString();
150            }
151            
152            
153            /**
154             * Parses a prompt set from the specified string.
155             *
156             * @param s The string to parse, must contain one or more 
157             *          non-conflicting space-delimited prompt types.
158             *
159             * @throws ParseException If the string couldn't be parsed to a valid
160             *                        prompt set.
161             */
162            public static Prompt parse(final String s)
163                    throws ParseException {
164            
165                    Prompt prompt = new Prompt();
166                    
167                    String[] tokens = s.split("\\s+");
168                    
169                    for (String t: tokens)
170                            prompt.add(Prompt.Type.parse(t));
171                    
172                    if (! prompt.isValid())
173                            throw new ParseException("Couldn't parse prompt: Invalid set");
174                    
175                    return prompt;
176            }
177    }
