061            /**
062             * Creates a new authorisation response. It must then be set with the
063             * requested {@link #setAuthorizationCode code}, 
064             * {@link #setAccessToken UserInfo access token} and / or 
065             * {@link #setIDToken ID token}.
066             *
067             * @param redirectURI The requested redirect URI. Must not be 
068             *                    {@code null}.
069             */
070            public AuthorizationResponse(final URL redirectURI) {
071            
072                    if (redirectURI == null)
073                            throw new NullPointerException("The redirect URI must not be null");
074                    
075                    this.redirectURI = redirectURI;
076            }

080             * Gets the requested redirect URI.
081             *
082             * @return The requested redirect URI.

090            /**
091             * Gets the corresponding response type set.
092             *
093             * @return The corresponding response type set.
094             */
095            public ResponseTypeSet getResponseTypeSet() {
096            
097                    ResponseTypeSet rts = new ResponseTypeSet();
098                    
099                    if (code != null)
100                            rts.add(ResponseType.CODE);
101                            
102                    if (idToken != null)
103                            rts.add(ResponseType.ID_TOKEN);
104                    
105                    if (accessToken != null)
106                            rts.add(ResponseType.TOKEN);
107                            
108                    return rts;
109            }

205            /**
206             * Returns the URL representation (redirect URI + fragment / query 
207             * string) of this authorisation response.
208             *
209             * <p>Example:
210             *
211             * <pre>
212             * https://client.example.com/cb?code=Qcb0Orv1zh30vL1MPRsbm&state=af0ifjsldkj
213             * </pre>
214             *
215             * @return The URL representation of this authorisation response.
216             *
217             * @throws IllegalStateException If there is no authorisation code, 
218             *                               UserInfo access token or ID token to
219             *                               serialise.
220             * @throws SerializeException    If this response couldn't be serialised
221             *                               to a URL.
222             */
223            public URL toURL()
224                    throws SerializeException {
225            
226                    if (code == null && idToken == null && accessToken == null)
227                            throw new IllegalStateException("Missing code, access token or ID token");
228            
229                    StringBuilder sb = new StringBuilder(redirectURI.toString());
230                    
231                    if (accessToken != null || idToken != null)
232                            sb.append('#');
233                    else
234                            sb.append('?');
235                    
236                    try {
237                            boolean delimit = false;
238                    
239                            if (code != null) {
240                                    sb.append("code=");
241                                    sb.append(URLEncoder.encode(code.getValue(), "utf-8"));
242                                    
243                                    delimit = true;
244                            }
245                            
246                            if (idToken != null) {
247                            
248                                    if (delimit)
249                                            sb.append('&');
250                                    
251                                    delimit = true;
252                            
253                                    sb.append("id_token=");
254                                    
255                                    try {
256                                            sb.append(idToken.serialize());
257                                            
258                                    } catch (JWTException e) {
259                                    
260                                            throw new SerializeException("Couldn't serialize ID token: " + e.getMessage(), e);
261                                    
262                                    }
263                            }
264                            
265                            if (accessToken != null) {
266                            
267                                    if (delimit)
268                                            sb.append('&');
269                                    
270                                    delimit = true;
271                            
272                                    sb.append("access_token=");
273                                    sb.append(URLEncoder.encode(accessToken.getValue(), "utf-8"));
274                                    sb.append("&token_type=");
275                                    sb.append(AccessToken.TYPE);
276                                    
277                                    final long exp = accessToken.getExpiration();
278                                    
279                                    if (exp > 0) {
280                                            sb.append("&expires_in=");
281                                            sb.append(exp);
282                                    }
283                                    
284                                    Scope scope = accessToken.getScope();
285                                    
286                                    if (scope != null) {
287                                            sb.append("&scope=");
288                                            sb.append(URLEncoder.encode(scope.toString(), "utf-8"));
289                                    }
290                            }
291                            
292                            if (state != null) {
293                            
294                                    if (delimit)
295                                            sb.append('&');
296                                    
297                                    sb.append("state=");
298                                    sb.append(URLEncoder.encode(state.toString(), "utf-8"));
299                            }
300                            
301                            return new URL(sb.toString());
302                    
303                    } catch (UnsupportedEncodingException e) {
304                    
305                            // UTF-8 should always be supported
306                            throw new SerializeException("Couldn't serialize response: " + e.getMessage(), e);
307                            
308                    } catch (MalformedURLException e) {
309                    
310                            throw new SerializeException("Couldn't serialize response: " + e.getMessage(), e);
311                    }
312            }
313

315            /**
316             * Returns the HTTP response for this authorisation response.
317             *
318             * <p>Example:
319             *
320             * <pre>
321             * HTTP/1.1 302 Found
322             * Location: https://client.example.com/cb?code=Qcb0Orv1zh30vL1MPRsbm&state=af0ifjsldkj
323             * </pre>
324             *
325             * @return The HTTP response matching this authorisation response.
326             *
327             * @throws SerializeException If the response couldn't be serialised to
328             *                            an HTTP response.
329             */
330            public HTTPResponse toHTTPResponse()
331                    throws SerializeException {
332            
333                    HTTPResponse response = new HTTPResponse(HTTPResponse.SC_FOUND);
334                    
335                    response.setLocation(toURL());
336                    
337                    return response;
338            }

341            /**
342             * Parses an authorisation response from the specified absolute or 
343             * relative URL.
344             *
345             * @param url The URL to parse. May be absolute or relative, with a
346             *            fragment or query string containing the authorisation
347             *            response parameters. Must not be {@code null}.
348             *
349             * @return The authorisation response.
350             *
351             * @throws ParseException If no valid authorisation response parameters
352             *                        are found in the URL.
353             */
354            public static AuthorizationResponse parse(final URL url)
355                    throws ParseException {
356                    
357                    if (url == null)
358                            throw new NullPointerException("The URL must not be null");
359                    
360                    String paramString = null;
361                    
362                    try {
363                            if (url.getQuery() != null)
364                                    paramString = URLDecoder.decode(url.getQuery(), "utf-8");
365                                    
366                            else if (url.getRef() != null)
367                                    paramString = URLDecoder.decode(url.getRef(), "utf-8");
368                            else
369                                    throw new ParseException("Missing authorization response parameters");
370                            
371                    } catch (UnsupportedEncodingException e) {
372                    
373                            // UTF-8 should always be supported
374                            throw new ParseException("Couldn't decode URL: " + e.getMessage(), e);
375                    }
376                            
377                    
378                    Map<String,String> params = URLUtils.parseParameters(paramString);
379                    
380                    if (params == null)
381                            throw new ParseException("Missing or invalid authorization response parameters");
382                    
383                    AuthorizationResponse response = new AuthorizationResponse(url);
384                    
385                    // Parse code parameter
386                    if (params.get("code") != null)
387                            response.setAuthorizationCode(new AuthorizationCode(params.get("code")));
388                    
389                    // Parse id_token parameter
390                    if (params.get("id_token") != null) {
391                    
392                            JWT idToken = null;
393                            
394                            try {
395                                    idToken.parse(params.get("id_token"));
396                                    
397                            } catch (JWTException e) {
398                            
399                                    throw new ParseException("Invalid ID Token JWT: " + e.getMessage(), e);
400                            }
401                            
402                            response.setIDToken(idToken);
403                    }
404                    
405                    // Parse access_token parameters
406                    if (params.get("access_token") != null) {
407                    
408                            AccessToken accessToken = new AccessToken(params.get("access_token"));
409                            
410                            if (params.get("expires_in") != null) {
411                            
412                                    long exp = -1;
413                            
414                                    try {
415                                            exp = new Long(params.get("expires_in"));
416                                            
417                                    } catch (NumberFormatException e) {
418                                    
419                                            throw new ParseException("Invalid expiration time: " + e.getMessage(), e);
420                                    }
421                                    
422                                    accessToken.setExpiration(exp);
423                            }
424                            
425                            if (params.get("scope") != null) {
426                            
427                                    try {
428                                            accessToken.setScope(Scope.parseStrict(params.get("scope")));
429                                            
430                                    } catch (ParseException e) {
431                                    
432                                            throw new ParseException("Invalid UserInfo scope: " + e.getMessage(), e);
433                                    }
434                            }
435                            
436                            response.setAccessToken(accessToken);
437                    }
438                    
439                    
440                    // Parse state parameter
441                    if (params.get("state") != null) {
442                    
443                            response.setState(new State(params.get("state")));
444                    }
445                    
446                    return response;
447            }
