001    package com.nimbusds.openid.connect.messages;
002    
003    
004    
005    /**
006     * OAuth 2.0 authorisation code.
007     *
008     * <p>See http://tools.ietf.org/html/draft-ietf-oauth-v2-25#section-1.3.1
009     *
010     * @author Vladimir Dzhuvinov
011     * @version 0.9 (2012-04-25)
012     */
013    public class AuthorizationCode {
014    
015    
016            /**
017             * The code value.
018             */
019            private String value;
020            
021            
022            /**
023             * Creates a new authorisation code.
024             *
025             * @param value The code value. Must not be {@code null} or empty 
026             *              string.
027             *
028             * @throws IllegalArgumentException If the code value is {@code null} or
029             *                                  empty string.
030             */
031            public AuthorizationCode(final String value) {
032            
033                    if (value == null || value.trim().isEmpty())
034                            throw new IllegalArgumentException("The authorization code value must not be null or empty string");
035                    
036                    this.value = value;
037            }
038            
039            
040            /**
041             * Gets the value of this authorisation code.
042             *
043             * @return The value.
044             */
045            public String getValue() {
046            
047                    return value;
048            }
049            
050            
051            /**
052             * Gets the string representation of this authorisation code.
053             *
054             * <p> See {@link #getValue}.
055             *
056             * @return The authorisation code value.
057             */
058            public String toString() {
059            
060                    return value;
061            }
062    }
