001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.net.URLEncoder;
005    import java.util.Map;
006    
007    import com.nimbusds.jwt.JWT;
008    import com.nimbusds.jwt.JWTException;
009    
010    import com.nimbusds.openid.connect.ParseException;
011    import com.nimbusds.openid.connect.SerializeException;
012    
013    import com.nimbusds.openid.connect.http.HTTPRequest;
014    import com.nimbusds.openid.connect.http.CommonContentTypes;
015    
016    import com.nimbusds.openid.connect.util.URLUtils;
017    
018    
019    /**
020     * Check ID request. To request the defails of the authentication performed on 
021     * the end-user, a request is made to the Check ID Endpoint sending the ID Token
022     * as the {@code access_token} by using the OAuth 2.0 Bearer scheme. 
023     *
024     * <p>The request can be constructed from a JSON Web Token (JWT) object or an 
025     * opaque string representing the ID Token to check.
026     *
027     * <p>Example HTTP request:
028     *
029     * <pre>
030     * POST /check_id HTTP/1.1
031     * Host: server.example.com
032     * Content-Type: application/x-www-form-urlencoded
033     * 
034     * access_token=eyJ0eXAiOiJKV1QiLCJhbGciOiJIUzI1NiJ9.eyJpc3MiOiJodHRwOlwvXC
035     * 9zZXJ2ZXIuZXhhbXBsZS5jb20iLCJ1c2VyX2lkIjoiMjQ4Mjg5NzYxMDAxIiwiYXVkIjoiaH
036     * R0cDpcL1wvY2xpZW50LmV4YW1wbGUuY29tIiwiZXhwIjoxMzExMjgxOTcwfQ.eDesUD0vzDH
037     * 3T1G3liaTNOrfaeWYjuRCEPNXVtaazNQ
038     * </pre>
039     *
040     * <p>See http://openid.net/specs/openid-connect-standard-1_0.html#anchor15
041     * <p>See http://openid.net/specs/openid-connect-messages-1_0.html#anchor10
042     *
043     * @author Vladimir Dzhuvinov
044     * @version 0.9 (2012-05-13)
045     */
046    public class CheckIDRequest implements Request {
047            
048             
049            /**
050             * The ID Token as an opaque access token.
051             */
052            private AccessToken accessToken = null;
053            
054            
055            /**
056             * The ID Token as JSON Web Token (JWT).
057             */
058            private JWT jwt = null;
059            
060            
061            /**
062             * Creates a new check ID request with the specified opaque string as 
063             * the access token.
064             *
065             * @param accessToken The access token, which should parse to a valid 
066             *                    JWT. Must not be {@code null}.
067             */
068            public CheckIDRequest(final AccessToken accessToken) {
069            
070                    if (accessToken == null)
071                            throw new NullPointerException("The access token must not be null");
072            
073                    this.accessToken = accessToken;
074            }
075            
076            
077            /**
078             * Creates a new check ID request with the specified JSON Web Token 
079             * (JWT) as the access token.
080             *
081             * @param jwt The access token, which must be a valid JWT. Must not be
082             *            {@code null}.
083             */
084            public CheckIDRequest(final JWT jwt) {
085            
086                    if (jwt == null)
087                            throw new NullPointerException("The JWT must not be null");
088            
089                    this.jwt = jwt;
090            }
091            
092            
093            /**
094             * Gets the access token as an opaque string.
095             *
096             * @return The access token.
097             *
098             * @throws JWTException If the access token was specified as a JWT which 
099             *                      is in a state doesn't permit serialisation (e.g. 
100             *                      unsigned or enencrypted).
101             */
102            public AccessToken getAccessToken()
103                    throws JWTException {
104            
105                    if (accessToken != null)
106                            return accessToken;
107                    else
108                            return new AccessToken(jwt.serialize());
109            }
110            
111            
112            /**
113             * Gets the access token as a JSON Web Token (JWT).
114             *
115             * @return The access token.
116             *
117             * @throws JWTException If the access token was specified as an opaque
118             *                      string which cannot be parsed to a JSON Web 
119             *                      Token (JWT).
120             */
121            public JWT getJWT()
122                    throws JWTException {
123    
124                    if (jwt != null)
125                            return jwt;
126                    else
127                            return JWT.parse(accessToken.getValue());
128            }
129            
130            
131            /**
132             * Returns the HTTP request for this check ID request.
133             *
134             * @return The HTTP request.
135             *
136             * @throws SerializeException If this check ID request couldn't be 
137             *                            serialised to an HTTP request.
138             */
139            public HTTPRequest toHTTPRequest()
140                    throws SerializeException {
141            
142                    HTTPRequest httpRequest = new HTTPRequest(HTTPRequest.Method.POST);
143                    
144                    httpRequest.setContentType(CommonContentTypes.APPLICATION_URLENCODED);
145                    
146                    StringBuilder sb = new StringBuilder("access_token=");
147                    
148                    if (accessToken != null) {
149                            sb.append(accessToken.getValue());
150                    }
151                    else {
152                            try {
153                                    sb.append(jwt.serialize());
154                                    
155                            } catch (JWTException e) {
156                            
157                                    throw new SerializeException("Couldn't serialize check ID request: " + e.getMessage(), e);
158                            }
159                    }
160            
161                    httpRequest.setQuery(sb.toString());
162            
163                    return httpRequest;
164            }
165            
166            
167            /**
168             * Parses a check ID request from the specified HTTP request. The access
169             * token is not checked for being a valid JSON Web Token (JWT).
170             *
171             * @param httpRequest The HTTP request. Must not be {@code null}.
172             *
173             * @return The parsed check ID request.
174             *
175             * @throws ParseException If the HTTP request couldn't be parsed to a 
176             *                        valid check ID request.
177             */
178            public static CheckIDRequest parse(final HTTPRequest httpRequest)
179                    throws ParseException {
180                    
181                    if (httpRequest.getContentType() != null &&
182                        ! httpRequest.getContentType().equals(CommonContentTypes.APPLICATION_URLENCODED))
183                            throw new ParseException("The Content-Type header must be application/x-www-form-urlencoded");
184                    
185                    String query = httpRequest.getQuery();
186                    
187                    if (query == null || query.isEmpty())
188                            throw new ParseException("Missing HTTP POST body");
189                    
190                    Map<String,String> params = URLUtils.parseParameters(query);
191                    
192                    String jwtString = params.get("access_token");
193                    
194                    if (jwtString == null || jwtString.isEmpty())
195                            throw new ParseException("Missing access_token form parameter");
196                    
197                    return new CheckIDRequest(new AccessToken(jwtString));
198            }
199    }
