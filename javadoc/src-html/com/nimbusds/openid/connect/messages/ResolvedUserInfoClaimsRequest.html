009    /**
010     * The resolved UserInfo claims request. It is determined by listing the default
011     * claims for the specified UserInfo {@link Scope} and then merging the UserInfo 
012     * claims from the optional {@link AuthorizationRequest#getResolvedRequestObject
013     * OpenID request object}.
014     *
015     * @author Vladimir Dzhuvinov
016     * @version 0.9 (2012-04-23)
017     */
018    public class ResolvedUserInfoClaimsRequest extends ClaimsRequest {
019            
020            
021            /**
022             * Gets the default UserInfo claims request for the
023             * {@link StdScopeMember#OPENID openid scope}.
024             *
025             * @return The UserInfo openid scope request.
026             */
027            public static JSONObject getClaimsRequestForOpenIDScope() {
028            
029                    JSONObject openidClaims = new JSONObject();
030                    openidClaims.put("user_id", null);
031                    return openidClaims;
032            }

034

055                    profileClaims.put("given_name", opt);
056                    
057                    opt = new JSONObject();
058                    opt.put("optional", true);
059                    profileClaims.put("middle_name", opt);
060    
061                    opt = new JSONObject();
062                    opt.put("optional", true);
063                    profileClaims.put("nickname", opt);
064    
065                    opt = new JSONObject();
066                    opt.put("optional", true);
067                    profileClaims.put("profile", opt);

187            
188            /**
189             * Gets the default claims for the specified UserInfo scope.
190             *
191             * @param scope The UserInfo scope. Must not be {@code null}.
192             *
193             * @return The matching UserInfo claims request.
194             */
195            public static JSONObject getClaimsForScope(final Scope scope) {
196            
197                    JSONObject claims = new JSONObject();
198                    
199                    if (scope.contains(StdScopeMember.OPENID))
200                            claims.putAll(getClaimsRequestForOpenIDScope());
201                            
202                    if (scope.contains(StdScopeMember.PROFILE))
203                            claims.putAll(getClaimsRequestForProfileScope());
204                    
205                    if (scope.contains(StdScopeMember.EMAIL))
206                            claims.putAll(getClaimsRequestForEmailScope());
207                            
208                    if (scope.contains(StdScopeMember.PHONE))
209                            claims.putAll(getClaimsRequestForPhoneScope());
210                    
211                    if (scope.contains(StdScopeMember.ADDRESS))
212                            claims.putAll(getClaimsRequestForAddressScope());
213                    
214                    return claims;
215            }
216            
217            
218            /**
219             * Creates a new resolved UserInfo claims request.
220             *
221             * @param scope          The UserInfo scope. Must not be {@code null}.
222             * @param userInfoObject The UserInfo object from the optional OpenID
223             *                       request object. {@code null} if not specified.
224             *
225             * @throws ResolveException If the ID Token claims request couldn't be
226             *                          resolved.
227             */
228            public ResolvedUserInfoClaimsRequest(final Scope scope, final JSONObject userInfoObject)
229                    throws ResolveException {
230                    
231                    // Resolve scope to claims
232                    claims.putAll(getClaimsForScope(scope));
233            
234                    if (userInfoObject != null) {
235                    
236                            if (userInfoObject.containsKey("claims") &&
237                                userInfoObject.get("claims") instanceof JSONObject) {
238                    
239                                    // Merge claims
240                                    JSONObject additionalClaims = (JSONObject)userInfoObject.get("claims");
241    
242                                    claims.putAll(additionalClaims);
243                            }
244                            
245                            try {
246                                    preferredLocales = ClaimsRequest.parsePreferredLocales(userInfoObject);
247                                    
248                            } catch (LangTagException e) {
249                            
250                                    throw new ResolveException("Couldn't parse preferred locales: " + e.getMessage(), e);
251                            }
252                    }
253            }
