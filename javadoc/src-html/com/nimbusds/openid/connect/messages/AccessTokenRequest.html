001    package com.nimbusds.openid.connect.messages;
002    
003    
004    import java.net.MalformedURLException;
005    import java.net.URL;
006    
007    import java.util.LinkedHashMap;
008    import java.util.Map;
009    
010    import com.nimbusds.openid.connect.ParseException;
011    import com.nimbusds.openid.connect.SerializeException;
012    
013    import com.nimbusds.openid.connect.http.CommonContentTypes;
014    import com.nimbusds.openid.connect.http.HTTPRequest;
015    
016    import com.nimbusds.openid.connect.util.URLUtils;
017    
018    
019    /**
020     * Access token request to the Token endpoint.
021     *
022     * <p>Example HTTP request (with private key JWT client authentication):
023     *
024     * <pre>
025     * POST /token HTTP/1.1
026     * Host: server.example.com
027     * Content-Type: application/x-www-form-urlencoded
028     * 
029     * grant_type=authorization_code&
030     * code=i1WsRn1uB1&
031     * client_id=s6BhdRkqt3&
032     * client_assertion_type=urn%3Aietf%3Aparams%3Aoauth%3Aclient-assertion-type%3Ajwt-bearer&
033     * client_assertion=PHNhbWxwOl...[omitted for brevity]...ZT
034     * </pre>
035     *
036     * <p>See http://tools.ietf.org/html/draft-ietf-oauth-v2-26#section-4.1.3
037     *
038     * <p>See draft-ietf-oauth-v2-26, section 4.1.3.
039     *
040     * @author Vladimir Dzhuvinov
041     * @version 0.9 (2012-05-11)
042     */
043    public class AccessTokenRequest extends TokenRequest {
044            
045            
046            /**
047             * The authorisation code received from the authorisation server.
048             */
049            private AuthorizationCode code;
050            
051            
052            /**
053             * The redirect URI.
054             */
055            private URL redirectURI;
056            
057            
058            /**
059             * Creates a new unauthenticated access token request with the specified
060             * parameters.
061             *
062             * @param code        The authorisation code received from the 
063             *                    authorisation server. Must not be {@code null}.
064             * @param redirectURI The redirect URI. Must not be {@code null}.
065             */
066            public AccessTokenRequest(final AuthorizationCode code, final URL redirectURI) {
067            
068                    this(code, redirectURI, null);
069            }
070            
071            
072            /**
073             * Creates a new authenticated access token request with the specified
074             * parameters.
075             *
076             * @param code        The authorisation code received from the 
077             *                    authorisation server. Must not be {@code null}.
078             * @param redirectURI The redirect URI. Must not be {@code null}.
079             * @param clientAuth  The client authentication, {@code null} if none.
080             */
081            public AccessTokenRequest(final AuthorizationCode code, final URL redirectURI, final ClientAuthentication clientAuth) {
082            
083                    super(GrantType.AUTHORIZATION_CODE, clientAuth);
084                    
085                    if (code == null)
086                            throw new NullPointerException("The authorization code must not be null");
087                    
088                    this.code = code;
089                    
090                    if (redirectURI == null)
091                            throw new NullPointerException("The redirect URI must not be null");
092                    
093                    this.redirectURI = redirectURI;
094            }
095            
096            
097            /**
098             * Gets the authorisation code.
099             *
100             * @return The authorisation code.
101             */
102            public AuthorizationCode getAuthorizationCode() {
103            
104                    return code;
105            }
106            
107            
108            /**
109             * Gets the redirect URI.
110             *
111             * @return The redirect URI.
112             */
113            public URL getRedirectURI() {
114            
115                    return redirectURI;
116            }
117            
118            
119            /**
120             * Returns the HTTP request for this access token request.
121             *
122             * @return The HTTP request.
123             *
124             * @throws SerializeException If this access token request couldn't be 
125             *                            serialised to an HTTP request.
126             */
127            public HTTPRequest toHTTPRequest()
128                    throws SerializeException {
129                    
130                    HTTPRequest httpRequest = new HTTPRequest(HTTPRequest.Method.POST);
131                    
132                    httpRequest.setContentType(CommonContentTypes.APPLICATION_URLENCODED);
133                    
134                    Map<String,String> params = new LinkedHashMap<String,String>();
135                    params.put("grant_type", getGrantType().toString());
136                    params.put("code", code.toString());
137                    params.put("redirect_uri", redirectURI.toString());
138                    
139                    httpRequest.setQuery(URLUtils.serializeParameters(params));
140                    
141                    if (getClientAuthentication() != null)
142                            getClientAuthentication().apply(httpRequest);
143                    
144                    return httpRequest;     
145            }
146            
147            
148            /**
149             * Parses the specified HTTP request for an access token request.
150             *
151             * @param httpRequest The HTTP request. Must not be {@code null}.
152             *
153             * @return The access token request.
154             *
155             * @throws ParseException If the HTTP request couldn't be parsed to a 
156             *                        valid access token request.
157             */
158            public static AccessTokenRequest parse(final HTTPRequest httpRequest)
159                    throws ParseException {
160                    
161                    // Only HTTP POST accepted
162                    httpRequest.ensureMethod(HTTPRequest.Method.POST);
163                    httpRequest.ensureContentType(CommonContentTypes.APPLICATION_URLENCODED);
164                    
165                    // No fragment!
166                    // May use query component!
167                    
168                    String query = httpRequest.getQuery();
169                    
170                    if (query == null)
171                            throw new ParseException("Missing HTTP POST request entity body");
172                    
173                    Map<String,String> params = URLUtils.parseParameters(query);
174                    
175                    
176                    // Parse grant type
177                    final String grantTypeString = params.get("grant_type");
178                    
179                    if (grantTypeString == null)
180                            throw new ParseException("Missing \"grant_type\" parameter");
181                            
182                    if (! grantTypeString.equals(GrantType.AUTHORIZATION_CODE.toString()))
183                            throw new ParseException("Invalid \"grant_type\" parameter: " + grantTypeString);
184                    
185                    
186                    // Parse authorisation code
187                    final String codeString = params.get("code");
188                    
189                    if (codeString == null)
190                            throw new ParseException("Missing \"code\" parameter");
191                    
192                    
193                    AuthorizationCode code = new AuthorizationCode(codeString);
194                    
195                    
196                    // Parse redirect URI
197                    final String redirectURIString = params.get("redirect_uri");
198                    
199                    if (redirectURIString == null) 
200                            throw new ParseException("Missing \"redirect_uri\" parameter");
201                    
202                    URL redirectURI = null;
203                    
204                    try {
205                            redirectURI = new URL(redirectURIString);
206                            
207                    } catch (MalformedURLException e) {
208                    
209                            throw new ParseException("Invalid \"redirect_uri\" parameter: " + e.getMessage(), e);
210                    }
211                    
212                    // Parse client authentication
213                    ClientAuthentication clientAuth = ClientAuthentication.parse(httpRequest);
214                    
215                    return new AccessTokenRequest(code, redirectURI, clientAuth);
216            }
217    }
