023    /**
024     * Authorisation error response.
025     *
026     * <p>Legal error codes:
027     *
028     * <ul>
029     *     <li>OAuth 2.0 errors:
030     *         <ul>
031     *             <li>{@link ErrorCode#INVALID_REQUEST}
032     *             <li>{@link ErrorCode#UNAUTHORIZED_CLIENT}
033     *             <li>{@link ErrorCode#ACCESS_DENIED}
034     *             <li>{@link ErrorCode#UNSUPPORTED_RESPONSE_TYPE}
035     *             <li>{@link ErrorCode#INVALID_SCOPE}
036     *             <li>{@link ErrorCode#SERVER_ERROR}
037     *             <li>{@link ErrorCode#TEMPORARILY_UNAVAILABLE}
038     *             <li>{@link ErrorCode#INVALID_REDIRECT_URI}
039     *         </ul>
040     *     <li>OpenID Connect specific errors:
041     *         <ul>
042     *             <li>{@link ErrorCode#INTERACTION_REQUIRED}
043     *             <li>{@link ErrorCode#LOGIN_REQUIRED}
044     *             <li>{@link ErrorCode#SESSION_SELECTION_REQUIRED}
045     *             <li>{@link ErrorCode#CONSENT_REQUIRED}
046     *             <li>{@link ErrorCode#INVALID_REQUEST_URI}
047     *             <li>{@link ErrorCode#INVALID_OPENID_REQUEST_OBJECT}
048     *         </ul>
049     * </ul>
050     *
051     * <p>Example HTTP response:
052     *
053     * <pre>
054     * HTTP/1.1 302 Found
055     * Location: https://client.example.com/cb?
056     * error=invalid_request
057     * &error_description=the%20request%20is%20not%20valid%20or%20malformed
058     * &state=af0ifjsldkj
059     * </pre>
060     *
061     * @author Vladimir Dzhuvinov
062     * @version 0.9 (2012-05-03)
063     */
064    public class AuthorizationErrorResponse implements ErrorResponse {
065    
066    
067            /**
068             * The legal error codes for an authorisation error response.
069             */
070            private static Set<ErrorCode> legalErrorCodes = new HashSet<ErrorCode>();
071            
072            
073            static {
074                    // OAuth 2.0 errors
075                    legalErrorCodes.add(ErrorCode.INVALID_REQUEST);
076                    legalErrorCodes.add(ErrorCode.UNAUTHORIZED_CLIENT);
077                    legalErrorCodes.add(ErrorCode.ACCESS_DENIED);
078                    legalErrorCodes.add(ErrorCode.UNSUPPORTED_RESPONSE_TYPE);
079                    legalErrorCodes.add(ErrorCode.INVALID_SCOPE);
080                    legalErrorCodes.add(ErrorCode.SERVER_ERROR);
081                    legalErrorCodes.add(ErrorCode.TEMPORARILY_UNAVAILABLE);
082                    
083                    // OpenID Connect specific errors
084                    legalErrorCodes.add(ErrorCode.INVALID_REDIRECT_URI);
085                    legalErrorCodes.add(ErrorCode.INTERACTION_REQUIRED);
086                    legalErrorCodes.add(ErrorCode.LOGIN_REQUIRED);
087                    legalErrorCodes.add(ErrorCode.SESSION_SELECTION_REQUIRED);
088                    legalErrorCodes.add(ErrorCode.CONSENT_REQUIRED);
089                    legalErrorCodes.add(ErrorCode.INVALID_REQUEST_URI);
090                    legalErrorCodes.add(ErrorCode.INVALID_OPENID_REQUEST_OBJECT);
091            }
092            
093            
094            /**
095             * Gets the legal error codes for an authorisation error response.
096             *
097             * @return The legal error codes, as a read-only set.
098             */
099            public static Set<ErrorCode> getLegalErrorCodes() {
100            
101                    return Collections.unmodifiableSet(legalErrorCodes);
102            }
103            
104            
105            /**
106             * The redirect URI.
107             */
108            private URL redirectURI;
109            
110             
111            /**
112             * The error code.
113             */
114            private ErrorCode errorCode;
115            
116            
117            /**
118             * The URL of a web page that includes additional information about the
119             * error.
120             */
121            private URL errorURI = null;
122            
123            
124            /**
125             * The response type set, used to determine redirect URL composition. If
126             * unknown {@code null}.
127             */
128            private ResponseTypeSet responseTypeSet;
129            
130            
131            /**
132             * The state parameter to be echoed back to the client.
133             */
134            private State state = null;
135            
136            
137            /**
138             * Creates a new authorisation error response.
139             *
140             * @param redirectURI     The redirect URI. Must not be {@code null}.
141             * @param errorCode       The error code. Must match one of the 
142             *                        {@link #getLegalErrorCodes legal error codes}
143             *                        for an authorisation error response and must 
144             *                        not be {@code null}.
145             * @param errorURI        Optional URI of a web page that includes 
146             *                        information about the error, {@code null} if 
147             *                        not specified.
148             * @param responseTypeSet The response type set, used to determine 
149             *                        redirect URL composition. If unknown
150             *                        {@code null}.
151             * @param state           The state parameter to be echoed back to the 
152             *                        client, {@code null} if not specified.
153             *
154             * @throws IllegalArgumentException If the specified error code is not
155             *                                  legal for an authorisation error 
156             *                                  response.
157             */
158            public AuthorizationErrorResponse(final URL redirectURI,
159                                              final ErrorCode errorCode,
160                                              final URL errorURI,
161                                              final ResponseTypeSet responseTypeSet,
162                                              final State state) {
163                                              
164                    if (redirectURI == null)
165                            throw new NullPointerException("The redirect URI must not be null");
166                    
167                    this.redirectURI = redirectURI;
168                    
169                    if (errorCode == null)
170                            throw new NullPointerException("The error code must not be null");
171                    
172                    if (! legalErrorCodes.contains(errorCode))
173                            throw new IllegalArgumentException("Illegal error code");
174                            
175                    this.errorCode = errorCode;
176                    
177                    this.errorURI = errorURI;
178                    
179                    this.responseTypeSet = responseTypeSet;
180                    
181                    this.state = state;
182            }

237            /**
238             * Returns the redirect URL with the appended error response parameters.
239             *
240             * @return The redirect URL with the appended error response parameters.
241             *
242             * @throws SerializeException If the redirect URL couldn't couldn't be
243             *                            produced.
244             */
245            public URL toURL()
246                    throws SerializeException {
247                    
248                    StringBuilder sb = new StringBuilder(redirectURI.toString());
249                    
250                    if (responseTypeSet == null ||
251                        responseTypeSet.contains(ResponseType.TOKEN) ||
252                        responseTypeSet.contains(ResponseType.ID_TOKEN))
253                            sb.append("#");
254                    else
255                            sb.append("?");
256                    
257                    try {
258                            sb.append("error=");
259                            sb.append(URLEncoder.encode(errorCode.getCode(), "utf-8"));
260    
261                            sb.append("&error_description=");
262                            sb.append(URLEncoder.encode(errorCode.getDescription(), "utf-8"));
263    
264                            if (errorURI != null) {
265                                    sb.append("&error_uri=");
266                                    sb.append(URLEncoder.encode(errorURI.toString(), "utf-8"));
267                            }
268    
269                            if (state != null) {
270                                    sb.append("&state=");
271                                    sb.append(URLEncoder.encode(state.toString(), "utf-8"));
272                            }
273                            
274                    } catch (UnsupportedEncodingException e) {
275                    
276                            throw new SerializeException("Couldn't serialize redirect URL: " + e.getMessage(), e);
277                    }
278                    
279                    try {
280                            return new URL(sb.toString());
281                            
282                    } catch (MalformedURLException e) {
283                    
284                            throw new SerializeException("Couldn't serialize redirect URL: " + e.getMessage(), e);
285                    }
286            }

289            /**
290             * @inheritDoc
291             */
292            public HTTPResponse toHTTPResponse()
293                    throws SerializeException {
294            
295                    HTTPResponse response = new HTTPResponse(HTTPResponse.SC_FOUND);
296                    
297                    response.setLocation(toURL());
298                    
299                    return response;
300            }

303            /**
304             * Parses an authorisation error response.
305             *
306             * @param url The redirect URL to parse. Must not be {@code null}.
307             *
308             * @throws ParseException If the redirect URL cannot be parsed to a 
309             *                        valid authorisation error response.
310             */
311            public static AuthorizationErrorResponse parse(final URL url)
312                    throws ParseException {
313                    
314                    Map<String,String> params = null;
315                    
316                    if (url.getRef() != null)
317                            params = URLUtils.parseParameters(url.getRef());
318                    else if (url.getQuery() != null)
319                            params = URLUtils.parseParameters(url.getQuery());
320                    else
321                            throw new ParseException("Missing URL reference or query string");
322                    
323                    // Parse the error code
324                    String errorCodeString = params.remove("error");
325                    ErrorCode errorCode = null;
326                    
327                    try {
328                            errorCode = ErrorCode.valueOf(errorCodeString.toUpperCase());
329                            
330                    } catch (NullPointerException e) {
331                    
332                            throw new ParseException("Missing error code");
333                    
334                    } catch (IllegalArgumentException e) {
335                    
336                            throw new ParseException("Invalid error code: " + errorCodeString);
337                    }
338                    
339                    // Ignore description
340                    params.remove("error_description");
341                    
342                    // Error page
343                    String urlString = params.remove("error_uri");
344                    URL errorURI = null;
345                    
346                    if (urlString != null) {
347                            
348                            try {
349                                    errorURI = new URL(urlString);
350                                    
351                            } catch (MalformedURLException e) {
352                    
353                                    throw new ParseException("Invalid error URI: " + urlString, e);
354                            }
355                    }
356                    
357                    // State
358                    String stateString = params.remove("state");
359                    State state = null;
360                    
361                    if (stateString != null)
362                            state = new State(stateString);
363                    
364                    // More params in URL?
365                    if (params.size() > 0)
366                            throw new ParseException("Unexpected parameter(s) in URL reference or query string");
367                    
368                    if (! getLegalErrorCodes().contains(errorCode))
369                            throw new ParseException("Illegal authorization response error code: " + errorCode.getCode());
370                    
371                    return new AuthorizationErrorResponse(URLUtils.getBaseURL(url), errorCode, errorURI, null, state);
372            }
373            
374            
375            /**
376             * Parses an authorisation error response.
377             *
378             * @param httpResponse The HTTP response to parse. Must not be 
379             *                     {@code null}.
380             *
381             * @throws ParseException If the HTTP response cannot be parsed to a 
382             *                        valid authorisation error response.
383             */
384            public static AuthorizationErrorResponse parse(final HTTPResponse httpResponse)
385                    throws ParseException {
386                    
387                    if (httpResponse.getStatusCode() != HTTPResponse.SC_FOUND)
388                            throw new ParseException("Unexpected HTTP status code, must be 302 (Found): " + httpResponse.getStatusCode());
389                    
390                    URL location = httpResponse.getLocation();
391                    
392                    if (location == null)
393                            throw new ParseException("Missing redirect URL / HTTP Location header");
394                    
395                    return parse(location);
396            }
